 21def gene_interaction_network(idata: pd.DataFrame, min_con: int = 2):
 22    """
 23    Creates a gene or protein interaction network graph.
 24
 25    The network is built from gene/protein interaction data. Nodes represent genes or proteins,
 26    edges represent interactions, and edge colors indicate the type of interaction.
 27
 28    Parameters
 29    ----------
 30    idata (pd.DataFrame): A DataFrame containing the interaction data with columns:
 31        - "A" (str): first gene/protein in the interaction
 32        - "B" (str): second gene/protein in the interaction
 33        - "connection_type" (str): interaction type, e.g., "gene -> protein"
 34
 35    min_con (int, optional): Minimum number of connections (node degree) required
 36                             for a gene/protein to be included in the network. Default is 2.
 37
 38    Returns
 39    -------
 40    nx.Graph: A NetworkX graph representing the interaction network.
 41        - Nodes have attributes:
 42            - "size": node size based on connection count (log-scaled)
 43            - "color": node color (default is 'khaki')
 44
 45        - Edges have attributes:
 46            - "color": edge color based on interaction type
 47
 48    Example
 49    -------
 50    >>> G = gene_interaction_network(interactions_df, min_con=3)
 51
 52    >>> nx.draw(G, with_labels=True, node_size=[G.nodes[n]['size'] for n in G.nodes()])
 53    """
 54
 55    inter = idata
 56    inter = inter[["A", "B", "connection_type"]]
 57
 58    dict_meta = pd.DataFrame(
 59        {
 60            "interactions": [
 61                ["gene -> gene"],
 62                ["protein -> protein"],
 63                ["gene -> protein"],
 64                ["protein -> gene"],
 65                ["gene -> gene", "protein -> protein"],
 66                ["gene -> gene", "gene -> protein"],
 67                ["gene -> gene", "protein -> gene"],
 68                ["protein -> protein", "gene -> protein"],
 69                ["protein -> protein", "protein -> gene"],
 70                ["gene -> protein", "protein -> gene"],
 71                ["gene -> gene", "protein -> protein", "gene -> protein"],
 72                ["gene -> gene", "protein -> protein", "protein -> gene"],
 73                ["gene -> gene", "gene -> protein", "protein -> gene"],
 74                ["protein -> protein", "gene -> protein", "protein -> gene"],
 75                [
 76                    "gene -> gene",
 77                    "protein -> protein",
 78                    "gene -> protein",
 79                    "protein -> gene",
 80                ],
 81            ],
 82            "color": [
 83                "#f67089",
 84                "#f47832",
 85                "#ca9213",
 86                "#ad9d31",
 87                "#8eb041",
 88                "#4fb14f",
 89                "#33b07a",
 90                "#35ae99",
 91                "#36acae",
 92                "#38a9c5",
 93                "#3aa3ec",
 94                "#957cf4",
 95                "#cd79f4",
 96                "#f35fb5",
 97                "#f669b7",
 98            ],
 99        }
100    )
101
102    genes_list = list(inter["A"]) + list(inter["B"])
103
104    genes_list = Counter(genes_list)
105
106    genes_list = pd.DataFrame(genes_list.items(), columns=["features", "n"])
107
108    genes_list = genes_list.sort_values("n", ascending=False)
109
110    genes_list = genes_list[genes_list["n"] >= min_con]
111
112    inter = inter[inter["A"].isin(list(genes_list["features"]))]
113    inter = inter[inter["B"].isin(list(genes_list["features"]))]
114
115    inter = inter.groupby(["A", "B"]).agg({"connection_type": list}).reset_index()
116
117    inter["color"] = "black"
118
119    for inx in inter.index:
120        for inx2 in dict_meta.index:
121            if set(inter["connection_type"][inx]) == set(
122                dict_meta["interactions"][inx2]
123            ):
124                inter["color"][inx] = dict_meta["color"][inx2]
125                break
126
127    G = nx.Graph()
128
129    for _, row in genes_list.iterrows():
130        node = row["features"]
131        color = "khaki"
132        weight = np.log2(row["n"] * 500)
133        G.add_node(node, size=weight, color=color)
134
135    for _, row in inter.iterrows():
136        source = row["A"]
137        target = row["B"]
138        color = row["color"]
139        G.add_edge(source, target, color=color)
140
141    return G

144def encrichment_cell_heatmap(
145    data: pd.DataFrame,
146    fig_size=(35, 25),
147    sets=None,
148    top_n=2,
149    test="FISH",
150    adj="BH",
151    parent_inc=False,
152    font_size=16,
153    clustering: str | None = "ward",
154    scale: bool = True,
155):
156    """
157    Creates a functional enrichment heatmap for cell types.
158
159    This function visualizes the most significant functional terms (GO, KEGG, REACTOME, specificity)
160    across different cell types.
161
162    Parameters
163    ----------
164    data (pd.DataFrame): Input data containing columns dependent on the source (GO-TERM, KEGG, REACTOME, specificity).
165
166    fig_size (tuple, optional): Figure size (width, height). Default is (35, 25).
167
168    sets (list, optional): List of specific cell sets to include. Default is None (all sets).
169
170    top_n (int, optional): Number of top terms to include per cell. Default is 2.
171
172    test (str, optional): Name of the statistical test column. Default is 'FISH'.
173
174    adj (str, optional): P-value adjustment method. Default is 'BH'.
175
176    parent_inc (bool, optional): Whether to include parent terms in labels. Default is False.
177
178    font_size (int, optional): Font size for the heatmap. Default is 16.
179
180    clustering (str | None, optional): Clustering method for rows/columns ('ward', 'single', None). Default is 'ward'.
181
182    scale (bool, optional): Whether to scale values before plotting. Default is True.
183
184    Returns
185    -------
186    matplotlib.figure.Figure: A heatmap figure of functional enrichment per cell type.
187
188    Raises
189    ------
190    ValueError: If the 'source' column in the data is not one of ['GO-TERM', 'KEGG', 'REACTOME', 'specificity'].
191
192    Example
193    -------
194    >>> fig = encrichment_cell_heatmap(data_df, top_n=3, parent_inc=True)
195
196    >>> fig.savefig('cell_heatmap.svg', bbox_inches='tight')
197    """
198
199    if not any(
200        x in data["source"].iloc[0]
201        for x in ("GO-TERM", "KEGG", "REACTOME", "specificity")
202    ):
203        raise ValueError(
204            "Invalid value for 'source' in data. Expected: 'GO-TERM', 'KEGG', 'REACTOME' or 'specificity'."
205        )
206
207    set_col = "cell"
208    if data["source"].iloc[0] == "GO-TERM":
209        term_col = "child_name"
210        parent_col = "parent_name"
211
212    elif data["source"].iloc[0] == "KEGG":
213        term_col = "3rd"
214        parent_col = "2nd"
215
216    elif data["source"].iloc[0] == "REACTOME":
217        term_col = "pathway"
218        parent_col = "top_level_pathway"
219
220    elif data["source"].iloc[0] == "specificity":
221        term_col = "specificity"
222        parent_col = "None"
223
224    title = f"Cells - {data['source'].iloc[0]}"
225
226    if isinstance(sets, list):
227        data[data["cell"].isin(sets)]
228
229    stat_col = [
230        x
231        for x in data.columns
232        if test in x and adj in x and parent_col.upper() not in x.upper()
233    ][0]
234
235    if parent_inc and data["source"].iloc[0] != "specificity":
236        data[term_col] = data.apply(
237            lambda row: f"{row[parent_col]} -> {row[term_col]}", axis=1
238        )
239
240    data = data.loc[data.groupby([set_col, term_col])[stat_col].idxmin()].reset_index(
241        drop=True
242    )
243
244    data = (
245        data.sort_values(stat_col, ascending=True).groupby(set_col).head(top_n)
246    ).reset_index(drop=True)
247
248    if sets is None and len(list(set(data["cell"]))) < 2:
249        if clustering is not None:
250            clustering = None
251            print(
252                "Clustering could not be conducted, because only one group is available in this analysis data."
253            )
254
255    figure = enrichment_heatmap(
256        data=data,
257        stat_col=stat_col,
258        term_col=term_col,
259        set_col=set_col,
260        sets=sets,
261        title=title,
262        fig_size=fig_size,
263        font_size=font_size,
264        scale=scale,
265        clustering=clustering,
266    )
267
268    return figure

271def draw_cell_conections(data, top_n=15):
272    """
273    Creates a cell-cell interaction network graph based on co-occurrence frequency.
274
275    The function generates a NetworkX graph where nodes represent cell types,
276    and edges represent the frequency of interactions between cells.
277
278    Parameters
279    ----------
280    data (pd.DataFrame): A DataFrame containing columns:
281        - "cell1" (str): source cell type
282        - "cell2" (str): target cell type
283
284    top_n (int, optional): Number of top interactions to include per source cell. Default is 15.
285
286    Returns
287    -------
288    nx.Graph: A NetworkX graph with attributes:
289        - Nodes:
290            - "size": node size (default 10)
291            - "color": node color (default "#FFA07A")
292
293        - Edges:
294            - "weight": edge weight (log-transformed from frequency)
295            - "color": edge color (default '#DCDCDC')
296            - "alpha": edge transparency (default 0.05)
297
298    Example
299    -------
300    >>> G = draw_cell_conections(cell_interactions_df, top_n=10)
301
302    >>> nx.draw(G, with_labels=True, node_size=[G.nodes[n]['size'] for n in G.nodes()])
303    """
304
305    cell_cell_df = (
306        data.groupby(["cell1", "cell2"])
307        .size()
308        .reset_index(name="weight")
309        .sort_values("weight", ascending=False)
310    )
311
312    cell_cell_df["weight"] = np.log(cell_cell_df["weight"])
313
314    cell_list = list(set(list(cell_cell_df["cell1"]) + list(cell_cell_df["cell2"])))
315
316    df_top = (
317        cell_cell_df.sort_values("weight", ascending=False).groupby("cell1").head(top_n)
318    )
319
320    G = nx.Graph()
321
322    for c in cell_list:
323        node = c
324        color = "#FFA07A"
325        weight = 10
326        G.add_node(node, size=weight, color=color)
327
328    for _, row in df_top.iterrows():
329        source = row["cell1"]
330        target = row["cell2"]
331        color = "#DCDCDC"
332        weight = row["weight"] / 10
333
334        G.add_edge(source, target, weight=weight, color=color, alpha=0.05)
335
336    nx.spring_layout(G, weight="weight", k=0.1, iterations=500)
337
338    return G

341def volcano_plot_conections(
342    deg_data: pd.DataFrame,
343    p_adj: bool = True,
344    top: int = 25,
345    top_rank: str = "p_value",
346    p_val: float | int = 0.05,
347    lfc: float | int = 0.25,
348    rescale_adj: bool = True,
349    image_width: int = 12,
350    image_high: int = 12,
351):
352    """
353    Generate a volcano plot from differential expression results.
354
355    A volcano plot visualizes the relationship between statistical significance
356    (p-values or standarized p-value) and log(fold change) for each gene, highlighting
357    genes that pass significance thresholds.
358
359    Parameters
360    ----------
361    deg_data : pandas.DataFrame
362        DataFrame containing differential expression results from calc_DEG() function.
363
364    p_adj : bool, default=True
365        If True, use adjusted p-values. If False, use raw p-values.
366
367    top : int, default=25
368        Number of top significant genes to highlight on the plot.
369
370    top_rank : str, default='p_value'
371        Statistic used primarily to determine the top significant genes to highlight on the plot. ['p_value' or 'FC']
372
373    p_val : float | int, default=0.05
374        Significance threshold for p-values (or adjusted p-values).
375
376    lfc : float | int, default=0.25
377        Threshold for absolute log fold change.
378
379    rescale_adj : bool, default=True
380        If True, rescale p-values to avoid long breaks caused by outlier values.
381
382    image_width : int, default=12
383        Width of the generated plot in inches.
384
385    image_high : int, default=12
386        Height of the generated plot in inches.
387
388    Returns
389    -------
390    matplotlib.figure.Figure: The generated volcano plot figure.
391    """
392
393    if top_rank.upper() not in ["FC", "P_VALUE"]:
394        raise ValueError("top_rank must be either 'FC' or 'p_value'")
395
396    if p_adj:
397        pv = "adj_pval"
398    else:
399        pv = "p_val"
400
401    deg_df = deg_data.copy()
402
403    shift = 0.25
404
405    p_val_scale = "-log(p_val)"
406
407    min_minus = min(deg_df[pv][(deg_df[pv] != 0) & (deg_df["log(FC)"] < 0)])
408    min_plus = min(deg_df[pv][(deg_df[pv] != 0) & (deg_df["log(FC)"] > 0)])
409
410    zero_p_plus = deg_df[(deg_df[pv] == 0) & (deg_df["log(FC)"] > 0)]
411    zero_p_plus = zero_p_plus.sort_values(by="log(FC)", ascending=False).reset_index(
412        drop=True
413    )
414    zero_p_plus[pv] = [
415        (shift * x) * min_plus for x in range(1, len(zero_p_plus.index) + 1)
416    ]
417
418    zero_p_minus = deg_df[(deg_df[pv] == 0) & (deg_df["log(FC)"] < 0)]
419    zero_p_minus = zero_p_minus.sort_values(by="log(FC)", ascending=True).reset_index(
420        drop=True
421    )
422    zero_p_minus[pv] = [
423        (shift * x) * min_minus for x in range(1, len(zero_p_minus.index) + 1)
424    ]
425
426    tmp_p = deg_df[
427        ((deg_df[pv] != 0) & (deg_df["log(FC)"] < 0))
428        | ((deg_df[pv] != 0) & (deg_df["log(FC)"] > 0))
429    ]
430
431    del deg_df
432
433    deg_df = pd.concat([zero_p_plus, tmp_p, zero_p_minus], ignore_index=True)
434
435    deg_df[p_val_scale] = -np.log10(deg_df[pv])
436
437    deg_df["top100"] = None
438
439    if rescale_adj:
440
441        deg_df = deg_df.sort_values(by=p_val_scale, ascending=False)
442
443        deg_df = deg_df.reset_index(drop=True)
444
445        eps = 1e-300
446        doubled = []
447        ratio = []
448        for n, i in enumerate(deg_df.index):
449            for j in range(1, 6):
450                if (
451                    n + j < len(deg_df.index)
452                    and (deg_df[p_val_scale][n] + eps)
453                    / (deg_df[p_val_scale][n + j] + eps)
454                    >= 2
455                ):
456                    doubled.append(n)
457                    ratio.append(
458                        (deg_df[p_val_scale][n + j] + eps)
459                        / (deg_df[p_val_scale][n] + eps)
460                    )
461
462        df = pd.DataFrame({"doubled": doubled, "ratio": ratio})
463        df = df[df["doubled"] < 100]
464
465        df["ratio"] = (1 - df["ratio"]) / 5
466        df = df.reset_index(drop=True)
467
468        df = df.sort_values("doubled")
469
470        if len(df["doubled"]) == 1 and 0 in df["doubled"]:
471            df = df
472        else:
473            doubled2 = []
474
475            for l in df["doubled"]:
476                if l + 1 != len(doubled) and l + 1 - l == 1:
477                    doubled2.append(l)
478                    doubled2.append(l + 1)
479                else:
480                    break
481
482            doubled2 = sorted(set(doubled2), reverse=True)
483
484        if len(doubled2) > 1:
485            df = df[df["doubled"].isin(doubled2)]
486            df = df.sort_values("doubled", ascending=False)
487            df = df.reset_index(drop=True)
488            for c in df.index:
489                deg_df.loc[df["doubled"][c], p_val_scale] = deg_df.loc[
490                    df["doubled"][c] + 1, p_val_scale
491                ] * (1 + df["ratio"][c])
492
493    deg_df.loc[(deg_df["log(FC)"] <= 0) & (deg_df[pv] <= p_val), "top100"] = "bisque"
494    deg_df.loc[(deg_df["log(FC)"] > 0) & (deg_df[pv] <= p_val), "top100"] = "skyblue"
495    deg_df.loc[deg_df[pv] > p_val, "top100"] = "lightgray"
496
497    if lfc > 0:
498        deg_df.loc[
499            (deg_df["log(FC)"] <= lfc) & (deg_df["log(FC)"] >= -lfc), "top100"
500        ] = "lightgray"
501
502    down_int = len(
503        deg_df["top100"][(deg_df["log(FC)"] <= lfc * -1) & (deg_df[pv] <= p_val)]
504    )
505    up_int = len(deg_df["top100"][(deg_df["log(FC)"] > lfc) & (deg_df[pv] <= p_val)])
506
507    deg_df_up = deg_df[deg_df["log(FC)"] > 0]
508
509    if top_rank.upper() == "P_VALUE":
510        deg_df_up = deg_df_up.sort_values([pv, "log(FC)"], ascending=[True, False])
511    elif top_rank.upper() == "FC":
512        deg_df_up = deg_df_up.sort_values(["log(FC)", pv], ascending=[False, True])
513
514    deg_df_up = deg_df_up.reset_index(drop=True)
515
516    n = -1
517    l = 0
518    while True:
519        n += 1
520        if deg_df_up["log(FC)"][n] > lfc and deg_df_up[pv][n] <= p_val:
521            deg_df_up.loc[n, "top100"] = "dodgerblue"
522            l += 1
523        if l == top or deg_df_up[pv][n] > p_val:
524            break
525
526    deg_df_down = deg_df[deg_df["log(FC)"] <= 0]
527
528    if top_rank.upper() == "P_VALUE":
529        deg_df_down = deg_df_down.sort_values([pv, "log(FC)"], ascending=[True, True])
530    elif top_rank.upper() == "FC":
531        deg_df_down = deg_df_down.sort_values(["log(FC)", pv], ascending=[True, True])
532
533    deg_df_down = deg_df_down.reset_index(drop=True)
534
535    n = -1
536    l = 0
537    while True:
538        n += 1
539        if deg_df_down["log(FC)"][n] < lfc * -1 and deg_df_down[pv][n] <= p_val:
540            deg_df_down.loc[n, "top100"] = "tomato"
541
542            l += 1
543        if l == top or deg_df_down[pv][n] > p_val:
544            break
545
546    deg_df = pd.concat([deg_df_up, deg_df_down])
547
548    que = ["lightgray", "bisque", "skyblue", "tomato", "dodgerblue"]
549
550    deg_df = deg_df.sort_values(
551        by="top100", key=lambda x: x.map({v: i for i, v in enumerate(que)})
552    )
553
554    deg_df = deg_df.reset_index(drop=True)
555
556    fig, ax = plt.subplots(figsize=(image_width, image_high))
557
558    plt.scatter(
559        x=deg_df["log(FC)"], y=deg_df[p_val_scale], color=deg_df["top100"], zorder=2
560    )
561
562    tl = deg_df[p_val_scale][deg_df[pv] >= p_val]
563
564    if len(tl) > 0:
565
566        line_p = np.max(tl)
567
568    else:
569        line_p = np.min(deg_df[p_val_scale])
570
571    plt.plot(
572        [max(deg_df["log(FC)"]) * -1.1, max(deg_df["log(FC)"]) * 1.1],
573        [line_p, line_p],
574        linestyle="--",
575        linewidth=3,
576        color="lightgray",
577        zorder=1,
578    )
579
580    if lfc > 0:
581        plt.plot(
582            [lfc * -1, lfc * -1],
583            [-3, max(deg_df[p_val_scale]) * 1.1],
584            linestyle="--",
585            linewidth=3,
586            color="lightgray",
587            zorder=1,
588        )
589        plt.plot(
590            [lfc, lfc],
591            [-3, max(deg_df[p_val_scale]) * 1.1],
592            linestyle="--",
593            linewidth=3,
594            color="lightgray",
595            zorder=1,
596        )
597
598    plt.xlabel("log(FC)")
599    plt.ylabel(p_val_scale)
600    plt.title("Volcano plot")
601
602    plt.ylim(min(deg_df[p_val_scale]) - 5, max(deg_df[p_val_scale]) * 1.3)
603
604    texts = [
605        ax.text(deg_df["log(FC)"][i], deg_df[p_val_scale][i], deg_df["feature"][i])
606        for i in deg_df.index
607        if deg_df["top100"][i] in ["dodgerblue", "tomato"]
608    ]
609
610    adjust_text(texts, arrowprops=dict(arrowstyle="-", color="gray", alpha=0.5))
611
612    legend_elements = [
613        Line2D(
614            [0],
615            [0],
616            marker="o",
617            color="w",
618            label="top-upregulated",
619            markerfacecolor="dodgerblue",
620            markersize=10,
621        ),
622        Line2D(
623            [0],
624            [0],
625            marker="o",
626            color="w",
627            label="top-downregulated",
628            markerfacecolor="tomato",
629            markersize=10,
630        ),
631        Line2D(
632            [0],
633            [0],
634            marker="o",
635            color="w",
636            label="upregulated",
637            markerfacecolor="skyblue",
638            markersize=10,
639        ),
640        Line2D(
641            [0],
642            [0],
643            marker="o",
644            color="w",
645            label="downregulated",
646            markerfacecolor="bisque",
647            markersize=10,
648        ),
649        Line2D(
650            [0],
651            [0],
652            marker="o",
653            color="w",
654            label="non-significant",
655            markerfacecolor="lightgray",
656            markersize=10,
657        ),
658    ]
659
660    ax.legend(handles=legend_elements, loc="upper right")
661    ax.grid(visible=False)
662
663    ax.annotate(
664        f"\nmin {pv} = " + str(p_val),
665        xy=(0.025, 0.975),
666        xycoords="axes fraction",
667        fontsize=12,
668    )
669
670    if lfc > 0:
671        ax.annotate(
672            "\nmin log(FC) = " + str(lfc),
673            xy=(0.025, 0.95),
674            xycoords="axes fraction",
675            fontsize=12,
676        )
677
678    ax.annotate(
679        "\nDownregulated: " + str(down_int),
680        xy=(0.025, 0.925),
681        xycoords="axes fraction",
682        fontsize=12,
683        color="black",
684    )
685
686    ax.annotate(
687        "\nUpregulated: " + str(up_int),
688        xy=(0.025, 0.9),
689        xycoords="axes fraction",

Generate a volcano plot from differential expression results.

A volcano plot visualizes the relationship between statistical significance (p-values or standarized p-value) and log(fold change) for each gene, highlighting genes that pass significance thresholds.

Parameters

deg_data : pandas.DataFrame DataFrame containing differential expression results from calc_DEG() function.

p_adj : bool, default=True If True, use adjusted p-values. If False, use raw p-values.

top : int, default=25 Number of top significant genes to highlight on the plot.

top_rank : str, default='p_value' Statistic used primarily to determine the top significant genes to highlight on the plot. ['p_value' or 'FC']

p_val : float | int, default=0.05 Significance threshold for p-values (or adjusted p-values).

lfc : float | int, default=0.25 Threshold for absolute log fold change.

rescale_adj : bool, default=True If True, rescale p-values to avoid long breaks caused by outlier values.

image_width : int, default=12 Width of the generated plot in inches.

image_high : int, default=12 Height of the generated plot in inches.

Returns

matplotlib.figure.Figure: The generated volcano plot figure.
